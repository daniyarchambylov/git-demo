git clone <REPO_ADDRESS> # Клонирует репозиторий
git log # Отображает историю репозитория / коммитов
git status # Отображает текущее положение проекта
git add # Добавляет изменения в файлах в репозиторий
git rm # Удаляет файл из репозитория
git commit -m '<Название коммита>' # Сохраняет изменения в локальном репозитории
git push origin master # синхронизирует локальный и удаленный репозитории
